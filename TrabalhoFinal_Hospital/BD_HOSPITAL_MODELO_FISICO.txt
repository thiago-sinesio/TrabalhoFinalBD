Criacao de Tabelas

CREATE TYPE status_leito AS ENUM ('ocupado', 'livre', 'manutencao');
CREATE TYPE turno_enfermeira AS ENUM ('manha', 'tarde', 'noite');
CREATE TYPE tipo_ala AS ENUM ('UTI', 'enfermaria', 'pediatria', 'cirurgia', 'maternidade');
CREATE TYPE tipo_atendimento AS ENUM ('consulta', 'emergencia', 'revisao');
CREATE TYPE status_atendimento AS ENUM ('realizado', 'cancelado', 'agendado');
CREATE TYPE tipo_exame AS ENUM ('sangue', 'imagem');
CREATE TYPE resultado_exame AS ENUM ('normal', 'alterado', 'critico');
CREATE TYPE tipo_laboratorio AS ENUM ('interno', 'externo');
CREATE TYPE status_fatura AS ENUM ('pendente', 'pago', 'cancelado', 'em_analise');
CREATE TYPE cobertura_plano AS ENUM ('regional', 'nacional');


CREATE TABLE hospital (
    id_hospital SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cnpj VARCHAR(20) NOT NULL UNIQUE
);

CREATE TABLE plano_saude (
    id_plano SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL UNIQUE,
    telefone VARCHAR(20),
    cobertura cobertura_plano NOT NULL
);

CREATE TABLE credenciamento_hospital_plano (
    id_credenciamento SERIAL PRIMARY KEY,
    id_hospital INT,
    id_plano INT,
    data_credenciamento DATE,
    FOREIGN KEY (id_hospital) REFERENCES hospital(id_hospital),
    FOREIGN KEY (id_plano) REFERENCES plano_saude(id_plano)
);

CREATE TABLE paciente (
    id_paciente SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cpf VARCHAR(14) UNIQUE,
    data_nascimento DATE,
    endereco TEXT,
    telefone VARCHAR(20),
    id_plano INT,
    FOREIGN KEY (id_plano) REFERENCES plano_saude(id_plano)
);

CREATE TABLE medico (
    id_medico SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    crm VARCHAR(20) NOT NULL UNIQUE,
    especialidade VARCHAR(100),
    telefone VARCHAR(20)
);

CREATE TABLE ala (
    id_ala SERIAL PRIMARY KEY,
    tipo tipo_ala NOT NULL,
    qtd_leitos_disponiveis INT,
    id_hospital INT,
    FOREIGN KEY (id_hospital) REFERENCES hospital(id_hospital)
);

CREATE TABLE enfermeira (
    id_enfermeira SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    cre VARCHAR(20) UNIQUE NOT NULL,
    turno turno_enfermeira NOT NULL,
    id_ala INT,
    id_chefe INT,
    FOREIGN KEY (id_ala) REFERENCES ala(id_ala),
    FOREIGN KEY (id_chefe) REFERENCES enfermeira(id_enfermeira)
);

CREATE TABLE leito (
    id_leito SERIAL PRIMARY KEY,
    numero VARCHAR(20) UNIQUE,
    status status_leito NOT NULL,
    id_ala INT,
    FOREIGN KEY (id_ala) REFERENCES ala(id_ala)
);

CREATE TABLE atendimento (
    id_atendimento SERIAL PRIMARY KEY,
    data_hora TIMESTAMP NOT NULL,
    tipo tipo_atendimento NOT NULL,
    status status_atendimento NOT NULL,
    observacoes TEXT,
    id_paciente INT,
    id_medico INT,
    FOREIGN KEY (id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY (id_medico) REFERENCES medico(id_medico)
);

CREATE TABLE medicamento (
    id_medicamento SERIAL PRIMARY KEY,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT
);

CREATE TABLE prescricao (
    id_prescricao SERIAL PRIMARY KEY,
    data_prescricao DATE DEFAULT CURRENT_DATE,
    dosagem VARCHAR(50),
    quantidade INT,
    id_atendimento INT,
    id_medicamento INT,
    FOREIGN KEY (id_atendimento) REFERENCES atendimento(id_atendimento) ON DELETE CASCADE,
    FOREIGN KEY (id_medicamento) REFERENCES medicamento(id_medicamento) ON DELETE CASCADE
);

CREATE TABLE internacao (
    id_internacao SERIAL PRIMARY KEY,
    id_paciente INT,
    id_leito INT,
    data_entrada TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    data_saida TIMESTAMP,
    FOREIGN KEY (id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY (id_leito) REFERENCES leito(id_leito),
    CONSTRAINT chk_datas CHECK (data_saida IS NULL OR data_saida > data_entrada) --Regra: Data saída tem que ser posterior a data entrada.
);

-- Regra: Paciente não pode possuir duas internações ativas.
CREATE UNIQUE INDEX idx_internacao_ativa ON internacao (id_paciente) WHERE data_saida IS NULL;

CREATE TABLE laboratorio (
    id_laboratorio SERIAL PRIMARY KEY,
    nome VARCHAR(100),
    tipo tipo_laboratorio,
    telefone VARCHAR(20)
);

CREATE TABLE exame (
    id_exame SERIAL PRIMARY KEY,
    id_paciente INT,
    id_medico INT,
    id_laboratorio INT,
    data_solicitacao TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    data_resultado TIMESTAMP,
    tipo tipo_exame, 
    custo DECIMAL(10,2) NOT NULL,
    descricao TEXT,
    resultado resultado_exame, 
    arquivo_laudo VARCHAR(255), -- Apenas a referência (caminho do arquivo)
    FOREIGN KEY (id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY (id_medico) REFERENCES medico(id_medico),
    FOREIGN KEY (id_laboratorio) REFERENCES laboratorio(id_laboratorio)
);

CREATE TABLE fatura (
    id_fatura SERIAL PRIMARY KEY,
    id_paciente INT,
    id_plano INT,
    id_atendimento INT,
    id_exame INT,
    id_internacao INT,
    valor DECIMAL(10,2) NOT NULL,
    status status_fatura DEFAULT 'pendente', 
    forma_pagamento VARCHAR(50),
    data_emissao DATE DEFAULT CURRENT_DATE,
    data_vencimento DATE NOT NULL,
    FOREIGN KEY (id_paciente) REFERENCES paciente(id_paciente),
    FOREIGN KEY (id_plano) REFERENCES plano_saude(id_plano),
    FOREIGN KEY (id_atendimento) REFERENCES atendimento(id_atendimento),
    FOREIGN KEY (id_exame) REFERENCES exame(id_exame),
    FOREIGN KEY (id_internacao) REFERENCES internacao(id_internacao),
    CONSTRAINT chk_origem_fatura CHECK (id_atendimento IS NOT NULL OR id_exame IS NOT NULL OR id_internacao IS NOT NULL)
);

CREATE TABLE pesquisa_satisfacao (
    id_pesquisa SERIAL PRIMARY KEY,
    id_atendimento INT UNIQUE,
    data_resposta TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    nota_geral INT CHECK (nota_geral BETWEEN 1 AND 5),
    comentario TEXT,
    recomendaria BOOLEAN NOT NULL DEFAULT FALSE,
    tempo_espera_avaliacao INT CHECK (tempo_espera_avaliacao BETWEEN 1 AND 5),
    FOREIGN KEY (id_atendimento) REFERENCES atendimento(id_atendimento)
);

Consulta 1:
select m.nome ,m.telefone
from medico m
where especialidade = 'Cardiologia'

Consulta 2: 
select p.nome,p.cpf,ps.nome
from paciente p 
inner join plano_saude ps on p.id_plano = ps.id_plano
where ps.nome = 'Unimed'

Consulta 3:
SELECT id_exame, tipo, resultado, data_solicitacao, data_resultado
FROM exame
WHERE data_resultado IS NULL
AND DATE_TRUNC('month', data_solicitacao) = DATE_TRUNC('month', CURRENT_DATE)

select l.nome ,count(e.id_exame)as quantidade
from exame e
inner join laboratorio l on e.id_laboratorio =l.id_laboratorio 
group by l.nome

Consulta 4;
select p.nome,l.id_leito,i.data_entrada
from internacao i
inner join paciente p on i.id_paciente = p.id_paciente 
inner join leito l on l.id_leito = i.id_leito 
where i.data_saida is null

Consulta 5;
select m.nome,a.data_hora,count(a.id_atendimento) as quantidade_de_atendimentos
from medico m
inner join atendimento a on m.id_medico = a.id_medico 
where a.data_hora between '2026-03-01' and '2026-03-31'
group by m.nome , a.data_hora 

SELECT m.nome, COUNT(a.id_atendimento) AS total
FROM medico m
JOIN atendimento a ON m.id_medico = a.id_medico
GROUP BY m.nome
ORDER BY total desc
LIMIT 1;

Consulta 6; 
SELECT a.id_ala,
       SUM(CASE 
             WHEN l.status = 'ocupado' THEN 1 
             ELSE 0 
           END)
           
           * 100.0 / COUNT(*) AS porcentagem_ocupação
FROM ala a
JOIN leito l ON a.id_ala = l.id_ala
GROUP BY a.id_ala;

Consulta 7;
select ps.nome ,f.data_emissao ,Sum(f.valor) as Valor_Total
from fatura f
inner join paciente p on f.id_paciente = p.id_paciente 
inner join plano_saude ps on ps.id_plano = p.id_plano 
WHERE f.data_emissao >= '2026-01-01'
  AND f.data_emissao < '2026-12-31'
  group by ps.nome , f.data_emissao

Consulta 8;
SELECT m.nome, COUNT(p.id_prescricao) AS total
FROM medicamento m
inner JOIN prescricao p ON m.id_medicamento = p.id_medicamento
GROUP BY m.nome
ORDER BY total DESC
limit 2;

consulta 9;
select m.nome,m.especialidade , count(a.id_paciente) quantidade_pacientes_atendidos_medicos
from medico m 
inner join atendimento a on m.id_medico = a.id_medico 
group by m.nome , m.especialidade


consulta 10;
SELECT l.numero AS numero_leito, p.nome AS nome_paciente, i.data_entrada
FROM internacao i
LEFT JOIN leito l ON l.id_leito = i.id_leito 
LEFT JOIN paciente p ON p.id_paciente = i.id_paciente 
WHERE i.data_saida IS NULL
AND i.data_entrada <= CURRENT_TIMESTAMP - INTERVAL '15 days';

consulta 11;
SELECT 
CASE
    WHEN id_atendimento IS NOT NULL THEN 'Atendimento'
    WHEN id_exame IS NOT NULL THEN 'Exame'
    WHEN id_internacao IS NOT NULL THEN 'Internacao'
    else 'Outro'
END AS tipo,
SUM(valor) AS total
FROM fatura
GROUP BY tipo;


consulta 12;
SELECT ps.nome, SUM(f.valor) AS total
FROM fatura f
JOIN paciente p ON f.id_paciente = p.id_paciente
JOIN plano_saude ps ON p.id_plano = ps.id_plano
WHERE ps.nome = 'Unimed'
GROUP BY ps.nome;
